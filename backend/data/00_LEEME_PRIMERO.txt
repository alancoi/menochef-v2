Backend Express:

  app.get('/api/recipes/:type', (req, res) => {
    const list = recipes.filter(r => r.type === req.params.type);
    res.json(list);
  });

Vue.js:

    recipesByType() {
      return recipes.filter(r => r.type === this.selectedType);
